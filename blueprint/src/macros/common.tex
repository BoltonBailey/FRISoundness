% In this file you should put all LaTeX macros to be used
% both by the pdf version and the web version.
% This should be most of your macros.

% Theorems
\theoremstyle{theorem}
\newtheorem{theorem}{Theorem}[section] % To be displayed in the blueprint
\newtheorem{thm}[theorem]{Theorem}     % Not to be displayed in the blueprint
\newtheorem{lemma}[theorem]{Lemma}     % To be displayed in the blueprint
\newtheorem{lem}[theorem]{Lemma}       % Not to be displayed in the blueprint
\newtheorem{proposition}[theorem]{Proposition} % To be displayed in the blueprint
\newtheorem{prop}[theorem]{Proposition}        % Not to be displayed in the blueprint
\newtheorem{corollary}[theorem]{Corollary}     % To be displayed in the blueprint
\newtheorem{cor}[theorem]{Corollary}           % Not to be displayed in the blueprint
\newtheorem{conjecture}[theorem]{Conjecture}   % To be displayed in the blueprint
\newtheorem{conj}[theorem]{Conjecture}         % Not to be displayed in the blueprint
\newtheorem{remark}[theorem]{Remark}           % To be displayed in the blueprint
\newtheorem{rmk}[theorem]{Remark}              % Not to be displayed in the blueprint
\newtheorem{observation}[theorem]{Observation} % To be displayed in the blueprint
\newtheorem{obs}[theorem]{Observation}         % To be displayed in the blueprint

% Definitions
\theoremstyle{definition}
\newtheorem{definition}[theorem]{Definition} % To be displayed in the blueprint
\newtheorem{defn}[theorem]{Definition}       % Not to be displayed in the blueprint
\newtheorem{example}[theorem]{Example}       % To be displayed in the blueprint
\newtheorem{ex}[theorem]{Example}            % Not to be displayed in the blueprint
\newtheorem{nex}[theorem]{Non-Example}
\newtheorem{ntn}[theorem]{Notation}
\newtheorem{con}[theorem]{Convention}
